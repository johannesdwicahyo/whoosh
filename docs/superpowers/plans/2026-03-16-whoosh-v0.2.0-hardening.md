# spec/integration/full_app_spec.rb
# frozen_string_literal: true

require "spec_helper"
require "rack/test"

class IntegrationUserSchema < Whoosh::Schema
  field :name,  String, required: true, desc: "User name"
  field :email, String, required: true, desc: "Email"
  field :age,   Integer, min: 0, max: 150
end

RSpec.describe "Full app integration" do
  include Rack::Test::Methods

  let(:application) do
    app = Whoosh::App.new

    app.auth do
      api_key header: "X-Api-Key", keys: { "sk-test" => { role: :standard } }
    end

    app.rate_limit do
      default limit: 100, period: 60
      rule "/limited", limit: 2, period: 60
    end

    app.access_control do
      role :standard, models: ["claude-haiku", "claude-sonnet"]
    end

    app.openapi do
      title "Integration Test API"
      version "1.0.0"
    end

    app.get "/health" do
      { status: "ok" }
    end

    app.post "/users", request: IntegrationUserSchema do |req|
      { created: true, name: req.body[:name] }
    end

    app.get "/users/:id" do |req|
      { id: req.params[:id] }
    end

    app.get "/protected", auth: :api_key do |req|
      { key: req.env["whoosh.auth"][:key] }
    end

    app.get "/limited" do
      { ok: true }
    end

    app.get "/events" do
      stream :sse do |out|
        out.event("ping", { ts: Time.now.to_i })
      end
    end

    app.post "/chat" do |req|
      stream_llm do |out|
        out << "Hello"
        out.finish
      end
    end

    app.post "/echo", mcp: true do |req|
      { echo: req.body }
    end

    app.group "/api/v1" do
      get "/items" do
        { items: [] }
      end
    end

    app.provide(:greeting) { "Hi" }
    app.get "/greet/:name" do |req, greeting:|
      { message: "#{greeting}, #{req.params[:name]}!" }
    end

    app
  end

  def app = application.to_rack

  # --- Basic routing ---

  it "serves simple GET" do
    get "/health"
    expect(last_response.status).to eq(200)
    expect(JSON.parse(last_response.body)["status"]).to eq("ok")
  end

  it "handles path params" do
    get "/users/42"
    expect(JSON.parse(last_response.body)["id"]).to eq("42")
  end

  it "returns 404 for unknown routes" do
    get "/nonexistent"
    expect(last_response.status).to eq(404)
  end

  it "sets JSON content-type" do
    get "/health"
    expect(last_response.content_type).to include("application/json")
  end

  # --- Schema validation ---

  it "validates and accepts valid input" do
    post "/users", { name: "Alice", email: "a@b.com", age: 25 }.to_json,
      "CONTENT_TYPE" => "application/json"
    expect(last_response.status).to eq(200)
    expect(JSON.parse(last_response.body)["name"]).to eq("Alice")
  end

  it "rejects invalid input with 422" do
    post "/users", {}.to_json, "CONTENT_TYPE" => "application/json"
    expect(last_response.status).to eq(422)
    body = JSON.parse(last_response.body)
    expect(body["error"]).to eq("validation_failed")
  end

  # --- Auth ---

  it "protects authenticated routes" do
    get "/protected"
    expect(last_response.status).to eq(401)
  end

  it "allows authenticated access" do
    get "/protected", {}, { "HTTP_X_API_KEY" => "sk-test" }
    expect(last_response.status).to eq(200)
    expect(JSON.parse(last_response.body)["key"]).to eq("sk-test")
  end

  # --- Rate limiting ---

  it "enforces rate limits" do
    3.times { get "/limited" }
    expect(last_response.status).to eq(429)
  end

  # --- Streaming ---

  it "streams SSE events" do
    get "/events"
    expect(last_response.status).to eq(200)
    expect(last_response.content_type).to eq("text/event-stream")
    expect(last_response.body).to include("event: ping")
  end

  it "streams LLM responses" do
    post "/chat"
    expect(last_response.body).to include("data: [DONE]")
  end

  # --- Groups ---

  it "routes grouped endpoints" do
    get "/api/v1/items"
    expect(last_response.status).to eq(200)
    expect(JSON.parse(last_response.body)["items"]).to eq([])
  end

  # --- DI ---

  it "injects dependencies" do
    get "/greet/World"
    expect(JSON.parse(last_response.body)["message"]).to eq("Hi, World!")
  end

  # --- Security headers ---

  it "includes security headers" do
    get "/health"
    expect(last_response.headers["x-content-type-options"]).to eq("nosniff")
    expect(last_response.headers["x-frame-options"]).to eq("DENY")
  end

  # --- OpenAPI ---

  it "serves OpenAPI spec" do
    get "/openapi.json"
    expect(last_response.status).to eq(200)
    spec = JSON.parse(last_response.body)
    expect(spec["openapi"]).to eq("3.1.0")
    expect(spec["info"]["title"]).to eq("Integration Test API")
    expect(spec["paths"]).to have_key("/health")
    expect(spec["paths"]).to have_key("/users")
    expect(spec["paths"]).to have_key("/users/{id}")
  end

  it "serves Swagger UI" do
    get "/docs"
    expect(last_response.status).to eq(200)
    expect(last_response.body).to include("swagger-ui")
  end

  # --- MCP ---

  it "registers MCP tools" do
    tools = application.mcp_server.list_tools
    expect(tools.map { |t| t[:name] }).to include("POST /echo")
  end
end

require "spec_helper"

RSpec.describe "Request edge cases" do
  def build_env(**opts)
    Rack::MockRequest.env_for("/test", **opts)
  end

  it "handles empty JSON body" do
    env = build_env(method: "POST", input: "", "CONTENT_TYPE" => "application/json")
    req = Whoosh::Request.new(env)
    expect(req.body).to be_nil
  end

  it "handles malformed JSON body" do
    env = build_env(method: "POST", input: "{not json", "CONTENT_TYPE" => "application/json")
    req = Whoosh::Request.new(env)
    expect { req.body }.to raise_error(JSON::ParserError)
  end

  it "handles binary body without content-type" do
    env = build_env(method: "POST", input: "\x00\x01\x02")
    req = Whoosh::Request.new(env)
    expect(req.body).to eq("\x00\x01\x02")
  end

  it "handles unicode path" do
    env = Rack::MockRequest.env_for("/users/caf%C3%A9")
    req = Whoosh::Request.new(env)
    expect(req.path).to eq("/users/caf%C3%A9")
  end

  it "handles empty query string" do
    env = Rack::MockRequest.env_for("/test?")
    req = Whoosh::Request.new(env)
    expect(req.query_params).to eq({})
  end

  it "handles duplicate query params" do
    env = Rack::MockRequest.env_for("/test?a=1&a=2")
    req = Whoosh::Request.new(env)
    # Rack::Utils.parse_query returns last value for duplicate keys
    expect(req.query_params["a"]).to be_a(String)
  end

  it "handles very long headers" do
    env = build_env(headers: {})
    env["HTTP_X_LONG"] = "x" * 10_000
    req = Whoosh::Request.new(env)
    expect(req.headers["X-Long"].length).to eq(10_000)
  end
end

# spec/edge_cases/router_edge_spec.rb
# frozen_string_literal: true

require "spec_helper"

RSpec.describe "Router edge cases" do
  let(:router) { Whoosh::Router.new }

  it "handles trailing slash" do
    router.add("GET", "/health", -> { "ok" })
    # /health/ should not match /health (no trailing slash normalization)
    expect(router.match("GET", "/health/")).to be_nil
  end

  it "handles root path" do
    handler = -> { "root" }
    router.add("GET", "/", handler)
    match = router.match("GET", "/")
    expect(match[:handler]).to eq(handler)
  end

  it "handles deeply nested paths" do
    handler = -> { "deep" }
    router.add("GET", "/a/b/c/d/e/f/g", handler)
    match = router.match("GET", "/a/b/c/d/e/f/g")
    expect(match[:handler]).to eq(handler)
  end

  it "handles overlapping param routes" do
    router.add("GET", "/users/:id", -> { "user" })
    router.add("GET", "/users/:id/posts", -> { "posts" })

    match = router.match("GET", "/users/42")
    expect(match[:params][:id]).to eq("42")

    match2 = router.match("GET", "/users/42/posts")
    expect(match2).not_to be_nil
  end

  it "handles special characters in path segments" do
    router.add("GET", "/files/:name", -> { "file" })
    match = router.match("GET", "/files/my-file.txt")
    expect(match[:params][:name]).to eq("my-file.txt")
  end

  it "returns nil for empty path" do
    router.add("GET", "/health", -> { "ok" })
    expect(router.match("GET", "")).to be_nil
  end
end

require "spec_helper"

RSpec.describe "Auth edge cases" do
  describe "API Key" do
    let(:auth) { Whoosh::Auth::ApiKey.new(keys: { "sk-valid" => { role: :basic } }, header: "X-Api-Key") }

    it "handles empty header value" do
      env = Rack::MockRequest.env_for("/test", "HTTP_X_API_KEY" => "")
      req = Whoosh::Request.new(env)
      expect { auth.authenticate(req) }.to raise_error(Whoosh::Errors::UnauthorizedError)
    end

    it "handles whitespace-only key" do
      env = Rack::MockRequest.env_for("/test", "HTTP_X_API_KEY" => "   ")
      req = Whoosh::Request.new(env)
      expect { auth.authenticate(req) }.to raise_error(Whoosh::Errors::UnauthorizedError)
    end
  end

  describe "JWT" do
    let(:auth) { Whoosh::Auth::Jwt.new(secret: "test-secret-32-chars-long!!!!!!!", algorithm: :hs256) }

    it "handles token with wrong number of segments" do
      env = Rack::MockRequest.env_for("/test", "HTTP_AUTHORIZATION" => "Bearer only.two")
      req = Whoosh::Request.new(env)
      expect { auth.authenticate(req) }.to raise_error(Whoosh::Errors::UnauthorizedError)
    end

    it "handles empty Bearer value" do
      env = Rack::MockRequest.env_for("/test", "HTTP_AUTHORIZATION" => "Bearer ")
      req = Whoosh::Request.new(env)
      expect { auth.authenticate(req) }.to raise_error(Whoosh::Errors::UnauthorizedError)
    end

    it "handles tampered payload" do
      token = auth.generate(sub: "user-1")
      parts = token.split(".")
      # Tamper with payload
      parts[1] = Base64.urlsafe_encode64('{"sub":"hacker","iat":9999999999,"exp":9999999999}', padding: false)
      tampered = parts.join(".")

      env = Rack::MockRequest.env_for("/test", "HTTP_AUTHORIZATION" => "Bearer #{tampered}")
      req = Whoosh::Request.new(env)
      expect { auth.authenticate(req) }.to raise_error(Whoosh::Errors::UnauthorizedError)
    end
  end

  describe "RateLimiter" do
    it "resets after window expires" do
      limiter = Whoosh::Auth::RateLimiter.new(default_limit: 2, default_period: 0.1)
      2.times { limiter.check!("key", "/test") }
      expect { limiter.check!("key", "/test") }.to raise_error(Whoosh::Errors::RateLimitExceeded)

      sleep 0.15
      expect { limiter.check!("key", "/test") }.not_to raise_error
    end
  end
end

require "spec_helper"

RSpec.describe "Middleware edge cases" do
  describe "middleware ordering" do
    it "applies middleware in correct order (outer first)" do
      order = []
      mw1 = Class.new { define_method(:initialize) { |app| @app = app }; define_method(:call) { |env| order << :first; @app.call(env) } }
      mw2 = Class.new { define_method(:initialize) { |app| @app = app }; define_method(:call) { |env| order << :second; @app.call(env) } }

      stack = Whoosh::Middleware::Stack.new
      stack.use(mw1)
      stack.use(mw2)
      inner = ->(env) { [200, {}, ["ok"]] }
      app = stack.build(inner)

      app.call(Rack::MockRequest.env_for("/"))
      expect(order).to eq([:first, :second])
    end
  end

  describe "RequestLimit" do
    it "allows GET requests without content-length" do
      inner = ->(env) { [200, {}, ["ok"]] }
      app = Whoosh::Middleware::RequestLimit.new(inner, max_bytes: 10)
      env = Rack::MockRequest.env_for("/test", method: "GET")
      status, _, _ = app.call(env)
      expect(status).to eq(200)
    end
  end

  describe "CORS" do
    it "handles missing Origin header gracefully" do
      inner = ->(env) { [200, {}, ["ok"]] }
      app = Whoosh::Middleware::Cors.new(inner)
      env = Rack::MockRequest.env_for("/test")
      status, headers, _ = app.call(env)
