```ruby
              input: JSON.generate(params), "CONTENT_TYPE" => "application/json")

            if handler_data[:endpoint_class]
              handler_data[:endpoint_class].new.call(request)
            elsif handler_data[:block]
              app_ref.instance_exec(request, &handler_data[:block])
            end
          }
        )
```

```ruby
# spec/whoosh/mcp/transport/sse_spec.rb
# frozen_string_literal: true

require "spec_helper"
require "stringio"

RSpec.describe Whoosh::MCP::Transport::SSE do
  let(:server) { Whoosh::MCP::Server.new }
  let(:transport) { Whoosh::MCP::Transport::SSE.new(server) }

  before do
    server.register_tool(
      name: "echo",
      description: "Echo input",
      handler: -> (params) { { echo: params["text"] } }
    )
  end

  describe "#handle_request" do
    it "handles JSON-RPC request and returns SSE response" do
      request_body = JSON.generate({
        jsonrpc: "2.0", method: "tools/list", id: 1
      })

      status, headers, body = transport.handle_request(request_body)
      expect(status).to eq(200)
      expect(headers["content-type"]).to eq("text/event-stream")
      response_data = body.first
      expect(response_data).to include("data:")

      # Parse the SSE data line
      json_line = response_data.lines.find { |l| l.start_with?("data:") }
      parsed = JSON.parse(json_line.sub("data: ", "").strip)
      expect(parsed["result"]["tools"].first["name"]).to eq("echo")
    end

    it "handles tools/call" do
      request_body = JSON.generate({
        jsonrpc: "2.0", method: "tools/call", id: 2,
        params: { name: "echo", arguments: { text: "hello" } }
      })

      status, _, body = transport.handle_request(request_body)
      expect(status).to eq(200)
      json_line = body.first.lines.find { |l| l.start_with?("data:") }
      parsed = JSON.parse(json_line.sub("data: ", "").strip)
      expect(parsed["result"]["content"].first["text"]).to include("hello")
    end

    it "returns error for invalid JSON" do
      status, _, body = transport.handle_request("not json")
      expect(status).to eq(200)
      json_line = body.first.lines.find { |l| l.start_with?("data:") }
      parsed = JSON.parse(json_line.sub("data: ", "").strip)
      expect(parsed["error"]).not_to be_nil
    end
  end
end
```

- [ ] **Step 2: Write implementation**

```ruby
# lib/whoosh/mcp/transport/sse.rb
# frozen_string_literal: true

module Whoosh
  module MCP
    module Transport
      class SSE
        def initialize(server)
          @server = server
        end

        def handle_request(body)
          begin
            request = Protocol.parse(body)
            response = @server.handle(request)
          rescue Protocol::ParseError => e
            response = Protocol.error_response(id: nil, code: -32700, message: e.message)
          end

          sse_body = if response
            "data: #{Protocol.encode(response)}\n\n"
          else
            ""
          end

          headers = {
            "content-type" => "text/event-stream",
            "cache-control" => "no-cache",
            "connection" => "keep-alive"
          }

          [200, headers, [sse_body]]
        end
      end
    end
  end
end
```

```ruby
      manager.register(:echo, command: "ruby -e \"STDIN.each_line { |l| r = JSON.parse(l); STDOUT.puts(JSON.generate({ jsonrpc: '2.0', id: r['id'], result: {} })); STDOUT.flush }\"")
```

Read `lib/whoosh/mcp/client_manager.rb`. Add:

```ruby
require "open3"
require "json"

# Add to initialize:
@pids = {}

# Add public methods:
def spawn_client(name)
  config = @configs[name]
  raise Errors::DependencyError, "Unknown MCP client: #{name}" unless config

  stdin, stdout, stderr, wait_thr = Open3.popen3(config[:command])
  pid = wait_thr.pid

  client = Client.new(stdin: stdin, stdout: stdout)

  @mutex.synchronize do
    @clients[name] = client
    @pids[name] = { pid: pid, thread: wait_thr, stderr: stderr }
  end

  client
end

def pids
  @mutex.synchronize { @pids.transform_values { |v| v[:pid] } }
end

# Update shutdown_all to kill processes:
def shutdown_all
  @mutex.synchronize do
    @clients.each_value { |c| c.close if c.respond_to?(:close) }
    @pids.each_value do |info|
      begin
        Process.kill("TERM", info[:pid])
        info[:thread]&.join(5)
      rescue Errno::ESRCH, Errno::ECHILD
        # Process already dead
      end
      info[:stderr]&.close rescue nil
    end
    @clients.clear
    @pids.clear
  end
end
```

```ruby
# spec/whoosh/mcp/stdio_e2e_spec.rb
# frozen_string_literal: true

require "spec_helper"
require "open3"

RSpec.describe "MCP stdio end-to-end" do
  it "communicates over stdio with a mock MCP server" do
    # Start a simple MCP server subprocess
    server_script = <<~RUBY
      require "json"
      STDOUT.sync = true
      STDIN.each_line do |line|
        req = JSON.parse(line)
        case req["method"]
        when "ping"
          STDOUT.puts JSON.generate({ jsonrpc: "2.0", id: req["id"], result: {} })
        when "tools/list"
          STDOUT.puts JSON.generate({ jsonrpc: "2.0", id: req["id"], result: { tools: [{ name: "greet", description: "Greet" }] } })
        when "tools/call"
          name = req.dig("params", "arguments", "name") || "World"
          STDOUT.puts JSON.generate({ jsonrpc: "2.0", id: req["id"], result: { content: [{ type: "text", text: "Hello, \#{name}!" }] } })
        else
          STDOUT.puts JSON.generate({ jsonrpc: "2.0", id: req["id"], error: { code: -32601, message: "Unknown" } })
        end
      end
    RUBY

    stdin, stdout, stderr, wait_thr = Open3.popen3("ruby", "-e", server_script)
    client = Whoosh::MCP::Client.new(stdin: stdin, stdout: stdout)

    # Ping
    expect(client.ping).to be true

    # Call tool
    result = client.call("greet", name: "Alice")
    expect(result[:content].first[:text]).to eq("Hello, Alice!")

    client.close
    stderr.close
    Process.kill("TERM", wait_thr.pid) rescue nil
    wait_thr.join(2)
  end
end
```
